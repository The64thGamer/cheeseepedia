{{- /* layouts/partials/build-photo-indexes.html Robust index builder for photos. - Always uses global `site`. - Ensures byPage entries are slices before append. - Dedupes by RelPermalink. - Returns dict { "byTitle","byPath","byPage","meta" }. */ -}} {{- $site := site -}} {{- if not $site -}} {{- return (dict "byTitle" (dict) "byPath" (dict) "byPage" (dict) "meta" (dict)) -}} {{- end -}} {{- $photos := where $site.RegularPages "Section" "photos" -}} {{- $byTitle := dict -}} {{- $byPath := dict -}} {{- $byPage := dict -}} {{- $meta := dict -}} {{- range $p := $photos -}} {{- $titleKey := $p.Title -}} {{- $filePathKey := $p.File.Path | default "" | printf "%s" -}} {{- /* Determine static file existence & URL */ -}} {{- $fileName := $titleKey -}} {{- $hasFile := false -}} {{- if ne $fileName "" -}} {{- $firstTry := printf "static/photos/%s" $fileName -}} {{- if fileExists $firstTry -}} {{- $hasFile = true -}} {{- else -}} {{- $exts := slice ".jpg" ".jpeg" ".png" ".webp" ".avif" ".gif" -}} {{- range $e := $exts -}} {{- $try := printf "static/photos/%s%s" $fileName $e -}} {{- if fileExists $try -}} {{- $hasFile = true -}} {{- $fileName = printf "%s%s" $fileName $e -}} {{- break -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $imageURL := cond $hasFile (printf "/photos/%s" $fileName) "/UI/File Not Found.jpg" -}} {{- /* store byTitle and byPath */ -}} {{- $byTitle = merge $byTitle (dict $titleKey $p) -}} {{- if ne $filePathKey "" -}} {{- $byPath = merge $byPath (dict $filePathKey $p) -}} {{- end -}} {{- /* store meta */ -}} {{- $meta = merge $meta (dict $titleKey (dict "imageURL" $imageURL "hasFile" $hasFile "fileName" $fileName)) -}} {{- /* Normalize .Params.pages into flat slice */ -}} {{- $pagesList := slice -}} {{- $rawPages := $p.Params.pages -}} {{- if $rawPages -}} {{- $t := printf "%T" $rawPages -}} {{- if in $t "string" -}} {{- $pagesList = slice $rawPages -}} {{- else if in $t "[]" -}} {{- range $el := $rawPages -}} {{- $et := printf "%T" $el -}} {{- if in $et "string" -}} {{- $pagesList = append $pagesList $el -}} {{- else if in $et "[]" -}} {{- range $inner := $el -}} {{- $pagesList = append $pagesList (printf "%v" $inner) -}} {{- end -}} {{- else -}} {{- $pagesList = append $pagesList (printf "%v" $el) -}} {{- end -}} {{- end -}} {{- else -}} {{- $pagesList = default (slice) $rawPages -}} {{- end -}} {{- end -}} {{- /* Merge into byPage with defensive conversion to slice & dedupe by RelPermalink */ -}} {{- range $pRef := $pagesList -}} {{- $key := printf "%s" $pRef -}} {{- /* Get raw existing value (might be nil, a Page, or a slice) */ -}} {{- $existingRaw := index $byPage $key -}} {{- /* Convert to a slice if needed */ -}} {{- $existing := slice -}} {{- if $existingRaw -}} {{- $type := printf "%T" $existingRaw -}} {{- if in $type "[]" -}} {{- /* existingRaw already a slice */ -}} {{- $existing = $existingRaw -}} {{- else -}} {{- /* single page stored â€” wrap it */ -}} {{- $existing = slice $existingRaw -}} {{- end -}} {{- end -}} {{- /* dedupe by RelPermalink */ -}} {{- $already := where $existing "RelPermalink" $p.RelPermalink -}} {{- if eq (len $already) 0 -}} {{- $byPage = merge $byPage (dict $key (append $existing $p)) -}} {{- end -}} {{- end -}} {{- end -}} {{- return (dict "byTitle" $byTitle "byPath" $byPath "byPage" $byPage "meta" $meta) -}}
